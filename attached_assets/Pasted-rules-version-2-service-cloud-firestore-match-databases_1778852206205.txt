rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAuth() { return request.auth != null; }
    function isOwner(uid) { return request.auth.uid == uid; }
    function isAdmin() {
      return isAuth() && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }

    match /users/{uid} {
      allow read: if isAuth();
      allow write: if isOwner(uid);
    }

    match /splits/{id} {
      allow read, update, delete: if isAuth() && resource.data.userId == request.auth.uid;
      allow create: if isAuth() && request.resource.data.userId == request.auth.uid;
    }

    match /workoutLogs/{id} {
      allow read, update, delete: if isAuth() && resource.data.userId == request.auth.uid;
      allow create: if isAuth() && request.resource.data.userId == request.auth.uid;
    }

    match /weightEntries/{id} {
      allow read, update, delete: if isAuth() && resource.data.userId == request.auth.uid;
      allow create: if isAuth() && request.resource.data.userId == request.auth.uid;
    }

    match /posts/{id} {
      allow read: if isAuth();
      allow create: if isAuth() && request.resource.data.userId == request.auth.uid;
      allow update: if isAuth() && (
        resource.data.userId == request.auth.uid ||
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['likesCount', 'likedBy'])
      );
      allow delete: if isAuth() && resource.data.userId == request.auth.uid;
    }

    match /follows/{id} {
      allow read: if isAuth();
      allow create: if isAuth() && request.auth.uid == request.resource.data.followerId;
      allow delete: if isAuth() && request.auth.uid == resource.data.followerId;
    }

    match /followRequests/{id} {
      allow read: if isAuth() && (
        resource.data.requesterId == request.auth.uid ||
        resource.data.targetId == request.auth.uid
      );
      allow create: if isAuth() && request.resource.data.requesterId == request.auth.uid;
      allow update: if isAuth() && resource.data.targetId == request.auth.uid;
    }

    match /blocks/{id} {
      allow read, write: if isAuth() && resource.data.blockerId == request.auth.uid;
      allow create: if isAuth() && request.resource.data.blockerId == request.auth.uid;
    }

    match /groups/{id} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update: if isAuth() && resource.data.memberIds.hasAny([request.auth.uid]);
      allow delete: if isAuth() && resource.data.createdBy == request.auth.uid;
    }

    match /reports/{id} {
      allow create: if isAuth() && request.resource.data.reporterId == request.auth.uid;
      allow read, write: if isAdmin();
    }
  }
}
